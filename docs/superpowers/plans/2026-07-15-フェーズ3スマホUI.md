function go(screenId, opts = {}) {
  const cur = document.querySelector(`.screen[data-screen="${state.screen}"]`);
  if (cur) cur.hidden = true;
  const next = document.querySelector(`.screen[data-screen="${screenId}"]`);
  if (!next) { console.warn('no screen', screenId); return; }
  if (!opts.back) state.history.push(state.screen);
  state.screen = screenId;
  next.hidden = false;
  const back = document.getElementById('backBtn');
  back.hidden = state.history.length === 0;
  window.scrollTo(0, 0);
}
function goBack() {
  if (!state.history.length) return;
  const prev = state.history.pop();
  const cur = document.querySelector(`.screen[data-screen="${state.screen}"]`);
  if (cur) cur.hidden = true;
  const p = document.querySelector(`.screen[data-screen="${prev}"]`);
  if (p) p.hidden = false;
  state.screen = prev;
  document.getElementById('backBtn').hidden = state.history.length === 0;
}

function makeChip(container, id, label, set) {
  const b = document.createElement('button');
  b.type = 'button'; b.className = 'chip'; b.textContent = label;
  b.setAttribute('aria-pressed', set.has(id) ? 'true' : 'false');
  b.addEventListener('click', () => {
    if (set.has(id)) { set.delete(id); b.setAttribute('aria-pressed', 'false'); b.classList.remove('on'); }
    else { set.add(id); b.setAttribute('aria-pressed', 'true'); b.classList.add('on'); }
  });
  container.appendChild(b);
}

function buildInputScreens() {
  // エリア（data_v2.regions が配列 or universities から抽出）

    : [...new Set(DATA.universities.map(u => u.region))].map(r => ({ id: r, label: r }));
  const rc = document.getElementById('regionChips');
  regionList.forEach(r => makeChip(rc, r.id, r.label || r.id, state.regions));

  const ic = document.getElementById('interestChips');

  const cc = document.getElementById('careerChips');

    examMethods: [...state.examMethods],
    riasecAnswers: riasecArr,
    budgetMax: null,
  };
}

function runDiagnose(riasecArr) {
  go('loading');
  const profile = userProfile(riasecArr);
  const r = ENGINE.diagnose(profile, DATA, DATA.tags, RIASEC_MAP);
  state.lastResult = r;
  renderResult(r);
  go('result');
}

function deptCard(item) {
  const el = document.createElement('div'); el.className = 'dept-card';
  const h = item.hensachi || {};
  const band = (h.min != null && h.max != null) ? `偏差${h.min}-${h.max}` : '偏差値データなし';
  el.innerHTML =
    `<div class="dept-top"><span class="dept-name">${item.uni_name} ${item.dept_name}</span>` +
    `<span class="fit-badge">${item.fitScore}</span></div>` +
    `<div class="dept-sub"><small>${band}</small></div>` +
    `<div class="reasons">${(item.reasons || []).map(x => `<span class="rchip">${x}</span>`).join('')}</div>`;
  el.addEventListener('click', () => showDetail(item.dept_id));
  return el;
}

function zoneSection(title, arr) {
  if (!arr || !arr.length) return null;
  const sec = document.createElement('div'); sec.className = 'zone';
  const h = document.createElement('h3'); h.textContent = `${title}（${arr.length}）`;
  sec.appendChild(h);
  arr.forEach(it => sec.appendChild(deptCard(it)));
  return sec;
}

function renderResult(r) {
  // サマリー
  const m = r.meta || {};
  const holland = m.hollandCode
    ? `${m.hollandCode}（${[...m.hollandCode].map(c => HOLLAND_JP[c] || c).join('・')}）` : '—';
  const summary = document.getElementById('resultSummary');
  summary.innerHTML =
    `<div class="summary-band">換算偏差値 ${m.adjustedHensachi ?? '—'} ｜ 適性 ${holland}</div>` +
    (m.lowSignal ? `<div class="nudge">適性チェックや興味選択をすると精度が上がります</div>` : '');

  const body = document.getElementById('resultBody');
  body.innerHTML = '';

  const z = r.zones || { challenge: [], match: [], safe: [] };
  const hasZones = (z.challenge.length + z.match.length + z.safe.length) > 0;

  if (m.adjustedHensachi == null || !hasZones) {
    // 偏差値未入力 or ゾーンが作れない → fit順1リスト（ungrouped）
    const list = (r.ungrouped && r.ungrouped.length) ? r.ungrouped
      : [...z.match, ...z.challenge, ...z.safe];
    if (!list.length) {
      body.innerHTML = `<p>条件に合う学部が見つかりませんでした。エリアや偏差値を見直してください。</p>` +
        `<small>除外: ${JSON.stringify(r.excluded || {})}</small>`;
      return;
    }
    const sec = zoneSection('おすすめ（適性順）', list);
    body.appendChild(sec);
    return;
  }

  [['🎯 実力相応', z.match], ['🚀 挑戦', z.challenge], ['🛡️ 安全', z.safe]].forEach(([t, a]) => {
    const s = zoneSection(t, a); if (s) body.appendChild(s);
  });
  // 偏差値欠損の混在分
  const ung = zoneSection('偏差値データなし', r.ungrouped);
  if (ung) body.appendChild(ung);
}

function findDept(deptId) {

function showDetail(deptId) {
  const { dept, uni } = findDept(deptId);
  if (!dept || !uni) return;

  const h = dept.hensachi || {};
  const bai = dept.bairitsu && dept.bairitsu.ippan ? dept.bairitsu.ippan : null;

    ['学費(年)', uni.tuition_yen_per_year ? `${uni.tuition_yen_per_year.toLocaleString()}円` : '—'],
  ];
  const links = (uni.official_links || []).map(l =>
    `<li><a href="${l.url}" target="_blank" rel="noopener">${l.label || l.url}</a></li>`).join('');
  document.getElementById('detailBody').innerHTML =
